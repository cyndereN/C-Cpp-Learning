#include<stdio.h>                               	/*包含头文件*/
int main()                                      	/*主函数main*/
{
    int num;                                    	/*定义变量*/
    printf(“please enter censorship:”);       		/*提示用户输入关卡*/
    scanf(“%d”,&num);                           	/*输入的数*/
    if(num==1)                                  	/*如果输入数字1*/
    {
        printf(“the current into first level\n”);	/*显示进入第一关*/ 
    }
    else if(num==2)                               	/*如果输入数字2*/
    {
        printf(“the current into second level\n”);	/*显示进入第二关*/
    }
    else if(num==3)                                	/*如果输入数字3*/ 
    {
        printf(“the current into third level\n”); 	/*显示进入第三关*/
    }
    else
    {
        printf(“the current into %d level\n”,num);	/*否则显示进入其他关*/
    }
      return 0;                                     	/*程序结束*/
}
