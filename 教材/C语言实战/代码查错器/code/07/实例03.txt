#include<stdio.h>                               	/*包含头文件*/
int main()                                      	/*主函数main*/
{
    	int num=0;                                	/*定义人数变量*/
    	scanf(“%d”,&num);                   		/*输入人数*/
     do                                     		/*进入循环*/
     {
         num++;                              		/*人数累加1*/
         printf(“还能承载 %d 人，\n”,26-num);           /*所剩座位*/
      }while(num<=25);                              	/*检验条件*/
      printf(“座位已满，不能再承载了。\n”);              /*输出信息*/
      return 0;                                     	/*程序结束*/
}
