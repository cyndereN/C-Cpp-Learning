#include<stdio.h>

int iGlobalPrice=10;					/*设定商店的初始价格*/

void Store1Price();					/*声明函数，代表1号连锁店*/
void Store2Price();					/*代表2号连锁店*/
void Store3Price();					/*代表3号连锁店*/
void ChangePrice();					/*更改连锁店的统一价格*/

int main()
{
		/*先显示价格改变之前所有连锁店手撕面包的价格*/
		printf(“手撕面包原价格是 : %d\n”,iGlobalPrice);
		Store1Price();					/*显示1号连锁店该产品的价格*/
		Store2Price();					/*显示2号连锁店该产品的价格*/
		Store3Price();					/*显示3号连锁店该产品的价格*/
		/*调用函数，改变连锁店手撕面包的价格*/
		ChangePrice();
		/*显示提示，显示修改后的价格*/
		printf(“手撕面包当前价格是: %d\n”,iGlobalPrice);
		Store1Price();					/*显示1号连锁店该产品的当前价格*/
		Store2Price();					/*显示2号连锁店该产品的当前价格*/
		Store3Price();					/*显示3号连锁店该产品的当前价格*/
		return 0;
}
/*定义1号连锁店手撕面包的价格函数*/
void Store1Price()
{
		printf(“1号连锁店手撕面包的价格是: %d\n”,iGlobalPrice);
}
/*定义2号连锁店手撕面包的价格函数*/
void Store2Price()
{
		printf(“2号连锁店手撕面包的价格是: %d\n”,iGlobalPrice);
}
/*定义3号连锁店手撕面包的价格函数*/
void Store3Price()
{
 		printf(“3号连锁店手撕面包的价格是: %d\n”,iGlobalPrice);
}
/*定义更改连锁店手撕面包的价格函数*/
void ChangePrice()
{
		printf(“手撕面包的价格调整后为: “);
		scanf(“%d”,&iGlobalPrice);
}
