#include<stdio.h>						/*包含头文件*/
void main()						/*主函数main*/
{
    unsigned result; 					/*定义无符号数*/
    int a, b; 						/*定义变量*/
    printf(“please input a:”); 				/*提示输入数据a*/
    scanf(“%d”,&a); 					/*输入数据a*/
    printf(“please input b:”); 				/*提示输入数据b*/
    scanf(“%d”,&b); 					/*输入数据b*/
    printf(“a=%d,b=%d”, a, b); 				/*显示数据a,b*/
    result = a^b; 					/*求a与b”异或”的结果*/
    printf(“\na^b=%u\n”, result); 			/*输出结果*/
}
